Titanium has a full API for adding database capabilities to your applications. Here's a simple example of how to use it.

// open db
var db = new ti.Database;
db.open('testdb');
// create table
db.execute("create table if not exists Test (value text)");
// insert row
db.execute("insert into Test values(?)",['test 1']);
// select data
var rs = db.execute("select value from Test");
// loop results
while(rs.isValidRow()) {
alert('retreieved value = ' + rs.field(0));
next();
}
// close results
rs.close();
db.close();

Titanium provides an API for dynamically creating native windows. Here is an example.

// create window object
var win = ti.Window.createWindow();
// set height
win.setHeight(600);
// set width
win.setWidth(800);
// set resizable attribute
win.setResizable(true);
// set URL
win.setURL('http://www.titaniumapp.com');
// open window
win.open();
// show window
win.show();

Titanium provides an API for dynamically creating system notifications. Here's an example.

// create notification object
var notification = new ti.Notification;
// set Title
notification.setTitle('My Title');
// set Message
notification.setMessage('It worked - pretty cool!');
// set Icon
notification.setIcon('app://images/logo.png');
// show
notification.show();

Titanium provides an API for dynamically creating a system tray icon. In this example, you can click the tray icon to hide and show the window. In Mac, it will appear in the top right, and in Windows it will appear at the bottom.

// set image (PR 1 does not support PNG in Win32 so we use .ico)
var trayIcon = (ti.platform == "win32")? "app://images/tray.ico":"app://images/tray.png";
// create tray menu with click callback
var menu = ti.Menu.createTrayMenu(trayIcon,null,function(sysmenu)
{
// if visible hide window else show
if (ti.Window.currentWindow.isVisible())
{
ti.Window.currentWindow.hide();
}
else
{
ti.Window.currentWindow.show();
}
});

Titanium provides an API for dynamically playing a sound. Here's an example.

// create sound object with url to audio file
var sound = ti.Media.createSound('app://audio/On.mp3');
// play
sound.play();

Titanium allows your app to open a full web browser using standard HTML anchors. Here's a simple example:

// use ti:systembrowser as your target
<a target="ti:systembrowser" href="http://www.appcelerator.org"></a>

Titanium provides an API for adding application menus. Here's an example where we add a menu under "Window".

// get menu object for "Window" menu item
var menu = ti.Menu.createAppMenu('Window');
// add custom item under "Window"
menu.addItem('My Menu', function()
{
// do nothing
});

Titanium provides an API for opening and reading files from the files system. Here's an example that opens a file dialog.

// open file dialog and count selected files
ti.Desktop.openFiles(function(files)
{
alert('files length ' + files.length)
});
